WorkoutSession

@Model
final class WorkoutSession {
    var id: UUID = UUID()
    var date: Date = Date()
    var note: String = ""

    @Relationship(deleteRule: .cascade, inverse: \WorkoutSet.session)
    var sets: [WorkoutSet] = []
}

Exercise

@Model
final class Exercise {
    var id: UUID = UUID()
    var name: String = ""
    var category: String = ""  // ExerciseCategory.rawValue
    var memo: String = ""
    var createdAt: Date = Date()
    var useCardioInput: Bool = false

    @Relationship(deleteRule: .cascade, inverse: \WorkoutSet.exercise)
    var sets: [WorkoutSet] = []

    var isCardio: Bool {
        category == ExerciseCategory.cardio.rawValue || useCardioInput
    }
}

/// 部位カテゴリ（v1.1.0: color プロパティ追加）
enum ExerciseCategory: String, CaseIterable, Identifiable {
    case chest      // 胸
    case back       // 背中
    case shoulders  // 肩
    case arms       // 腕
    case legs       // 脚
    case abs        // 腹筋
    case cardio     // 有酸素
    case other      // その他

    var displayName: String { ... }  // 日本語表示名

    /// 部位カラー（カレンダーヒートマップ・凡例で使用、v1.1.0）

        switch self {
        case .chest:     Color(red: 0.95, green: 0.35, blue: 0.40)  // 赤

WorkoutTemplate（v1.1.0: assignedWeekdays 追加）

@Model
final class WorkoutTemplate {
    var id: UUID = UUID()
    var name: String = ""
    var createdAt: Date = Date()

    /// 曜日固定ルーティン（v1.1.0）
    /// 0=日曜, 1=月曜, 2=火曜, 3=水曜, 4=木曜, 5=金曜, 6=土曜
    /// 空配列の場合は曜日固定なし（任意日のテンプレートとして使用可能）
    var assignedWeekdays: [Int] = []

    @Relationship(deleteRule: .cascade, inverse: \TemplateEntry.template)
    var entries: [TemplateEntry] = []

    var sortedEntries: [TemplateEntry] {
        entries.sorted { $0.order < $1.order }
    }

    /// 指定日（曜日基準）にこのルーティンが該当するか（v1.1.0）
    func appliesTo(date: Date) -> Bool {
        guard !assignedWeekdays.isEmpty else { return false }
        // Calendar.weekday: 1=Sunday...7=Saturday → 0=Sunday に正規化
        let weekday = Calendar.current.component(.weekday, from: date) - 1
        return assignedWeekdays.contains(weekday)
    }
}

TemplateEntry（v1.1.1: setEntries 追加）

@Model
final class TemplateEntry {
    var id: UUID = UUID()
    var exerciseName: String = ""
    var exerciseCategory: String = ""
    /// レガシーフィールド（後方互換。setEntries が正源）
    var weight: Double = 0
    var reps: Int = 0
    var sets: Int = 1
    var order: Int = 0
    var duration: Double = 0   // 有酸素: 分（レガシー）
    var distance: Double = 0   // 有酸素: km（レガシー）
    var isCardio: Bool = false

    /// v1.1.1: セットごとの値を保持する正源リレーション
    @Relationship(deleteRule: .cascade, inverse: \TemplateSetEntry.entry)
    var setEntries: [TemplateSetEntry] = []

    var template: WorkoutTemplate?

    var sortedSetEntries: [TemplateSetEntry] {
        setEntries.sorted { $0.order < $1.order }
    }
}

TemplateSetEntry（v1.1.1 新規）

/// セットごとの重量・回数・有酸素値を保持する（v1.1.1）
/// ピラミッドセット等の変則構成を表現できる
@Model
final class TemplateSetEntry {
    var id: UUID = UUID()
    var weight: Double = 0    // ウエイト種目: kg
    var reps: Int = 0         // ウエイト種目: 回
    var duration: Double = 0  // 有酸素種目: 分
    var distance: Double = 0  // 有酸素種目: km
    var order: Int = 0        // セットの並び順

    var entry: TemplateEntry?

    init(weight: Double = 0, reps: Int = 0, duration: Double = 0, distance: Double = 0, order: Int = 0) {
        self.weight = weight
        self.reps = reps
        self.duration = duration
        self.distance = distance
        self.order = order
    }
}

マイグレーション方針

SwiftDataの @Model は全プロパティにデフォルト値必須。新規プロパティ追加時は必ずデフォルト値を設定すること。

v1.1.0 マイグレーション: WorkoutTemplate.assignedWeekdays: [Int] = [] を追加。デフォルト空配列のため、既存データに影響なし（SwiftData lightweight migration で自動対応）。

v1.1.1 マイグレーション: TemplateEntry.setEntries: [TemplateSetEntry] = [] を追加。デフォルト空配列のため SwiftData lightweight migration で自動対応。既存の weight / reps / sets レガシーフィールドは保持（後方互換）。
起動時（WorkoutDiaryApp.onAppear）および EditTemplateView.onAppear で TemplateMigrationService.migrateAllIfNeeded() を呼び出し、既存 TemplateEntry を setEntries 形式に自動展開する（冪等）。
