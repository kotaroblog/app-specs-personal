StudySession

@Model final class StudySession {
    var id: UUID = UUID()
    var startTime: Date = Date()
    var endTime: Date?
    var duration: TimeInterval = 0
    var subject: Subject?

    var date: Date { Calendar.current.startOfDay(for: startTime) }
}

Subject

@Model final class Subject {
    var id: UUID = UUID()
    var name: String = ""
    var colorHex: String = "#007AFF"
    var iconName: String = "book"
    var sortOrder: Int = 0
    var createdAt: Date = Date()
    var currentScore: Int?
    var targetScore: Int?

    @Relationship(deleteRule: .cascade, inverse: \StudySession.subject)
    var sessions: [StudySession] = []
}

StudyTask

@Model final class StudyTask {
    var id: UUID = UUID()
    var periodRawValue: String = "day"   // TaskPeriod.rawValue: "day" / "week" / "month"
    var title: String = ""
    var isCompleted: Bool = false
    var periodStart: Date = Date()       // 期間開始（例: 今日0時）
    var periodEnd: Date = Date()         // 期間終了（例: 今日23:59:59）
    var linkedSessionID: UUID?           // 「記録して完了」時に紐づく StudySession.id
    var completedAt: Date?               // 完了日時
    var sortOrder: Int = 0
    var createdAt: Date = Date()
}

enum TaskPeriod: String {
    case day   = "day"
    case week  = "week"
    case month = "month"
}

TestRecord（v1.1.0 新規）

@Model final class TestRecord {
    var id: UUID = UUID()
    var date: Date = Date()
    var testTypeRaw: String = TestType.regular.rawValue  // TestType.rawValue を保存
    var subject: Subject? = nil                           // nullify リレーション
    var score: Double = 0.0
    var maxScore: Double = 100.0
    var deviation: Double? = nil   // 模試のみ使用
    var rank: Int? = nil           // 任意
    var memo: String? = nil        // 任意
    var createdAt: Date = Date()

    // Computed properties
    var scoreRate: Double { maxScore > 0 ? score / maxScore * 100 : 0 }
    var testType: TestType { TestType(rawValue: testTypeRaw) ?? .other }
}

enum TestType: String, Codable, CaseIterable {
    case regular = "regular"  // 定期テスト
    case mock    = "mock"     // 模試
    case quiz    = "quiz"     // 小テスト
    case other   = "other"    // その他

    var displayName: String {
        switch self {
        case .regular: return "定期テスト"
        case .mock:    return "模試"
        case .quiz:    return "小テスト"
        case .other:   return "その他"
        }
    }
}

Enum 一覧

  • GoalType: regularExam / highSchoolEntrance / universityEntrance
  • TaskPeriod: day / week / month（StudyTask の期間種別）
  • TestType: regular / mock / quiz / other（TestRecord のテスト種別）v1.1.0 新規
  • ShareCardType: daily / summary / heatmap / graph
  • ShareCardTheme: 複数テーマ（無料/プレミアム）
  • ShareSummaryPeriod: week / month
  • ShareGraphType: line / bar
  • OnboardingPage: welcome / goalType / subject / score / motivation / exam / target / commit / paywall
  • PomodoroPhase: focus（25分） / shortBreak（5分）

マイグレーション方針

SwiftDataの@Modelは全プロパティにデフォルト値必須。新規プロパティ追加時は必ずデフォルト値を設定する。
